<table class="dataTable">
	<colgroup>

		<col style="width: 20%;"/>

			<th class="center">번호</th>
			<th class="center">제목</th>

			<th class="center">작성날짜</th>
			<th class="center">조회수</th>
		</tr>
	</thead>
{{#each articleList}}
	<tbody>
		<tr>
			<td class="center">{{seq}}</td>
			<td>{{subject}}</td>
			<td>{{writerName}}</td>

			<td class="center">{{hit}}</td>
		</tr>
	</tbody>
{{/each}}

</script>
